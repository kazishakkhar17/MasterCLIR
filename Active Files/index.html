def precision_at_k(results, relevant_urls, k=10):
    top_k = results[:k]
    relevant_found = sum(1 for r in top_k if r['doc']['url'] in relevant_urls)
    return relevant_found / k

import pickle
import time
import streamlit as st

# Load Module C system
@st.cache_resource
def load_clir_system():
    with open('retrieval_system.pkl', 'rb') as f:
        return pickle.load(f)

# Module D: Add confidence
def get_confidence(score):
    if score >= 0.7: return "üü¢ High"
    elif score >= 0.4: return "üü° Medium"
    else: return "üî¥ Low"

# UI
st.title("üîç Bangla-English CLIR System")
query = st.text_input("Search (English or ‡¶¨‡¶æ‡¶Ç‡¶≤‡¶æ):")

if query:
    # Time the query
    start = time.time()
    results = hybrid.search(query, top_k=10)
    elapsed = (time.time() - start) * 1000
    
    # Display
    st.metric("Query Time", f"{elapsed:.2f} ms")
    st.metric("Results Found", len(results))
    
    if results:
        top_score = results[0]['score']
        confidence = get_confidence(top_score)
        st.metric("Confidence", confidence)
        
        if top_score < 0.2:
            st.warning("‚ö†Ô∏è Low confidence. Try rephrasing your query.")
        
        # Show results
        for i, r in enumerate(results, 1):
            with st.expander(f"[{i}] {r['doc']['title']} - Score: {r['score']:.3f}"):
                st.write(f"**Language:** {r['doc']['language'].upper()}")
                st.write(f"**Source:** {r['doc']['source']}")
                st.write(f"**Snippet:** {r['doc']['body'][:200]}...")
                st.link_button("Read Full Article", r['doc']['url'])
    else:
        st.error("No results found.")
